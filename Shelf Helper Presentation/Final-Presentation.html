Shelf Helper

Making Hard-to-Reach Products Accessible

A web-based solution for inclusive shopping experiences

Welcome to my presentation on the Shelf Helper application. Today I'll walk you through the development of this web-based solution designed to improve accessibility in retail environments.

Problem and Solution

The Problem

  • Many customers struggle to reach high-shelf products
  • Accessibility barriers for those with disabilities
  • Store staff not always available
  • Embarrassment when asking for help

Our Solution

  • Web application connecting customers with staff
  • Simple interface for browsing and requesting help
  • Accessible design for all users
  • Dignified way to get assistance
The core problem we're addressing is the difficulty many shoppers face when trying to reach items on high shelves. Our solution creates a dignified way for customers to request assistance without embarrassment or frustration.

    // Initialize SpeechRecognition with fallbacks

                            window.webkitSpeechRecognition;

      this.showError('Voice search not supported in this browser');
      return;
    }
    
    this.recognition = new SpeechRecognition();
    this.recognition.continuous = false;
    this.recognition.lang = 'en-US';
    this.recognition.interimResults = false;
    this.recognition.maxAlternatives = 3;
    
    this.setupEventListeners();
    this.loadRetailerSpecificDictionary();
  }

// Enterprise integration with Walmart/Costco inventory systems
class RetailIntegrationAPI {
  constructor(retailer, apiKey) {
    this.retailer = retailer; // 'walmart' or 'costco'
    this.apiKey = apiKey;
    this.baseUrl = this.getBaseUrl();
    this.headers = this.getDefaultHeaders();
  }

  getBaseUrl() {
    // Different endpoints for different retailers
    const endpoints = {
      walmart: 'https://api.walmart.retail.example.com/v2/',
      costco: 'https://api.costco.retail.example.com/v3/'
    };
    return endpoints[this.retailer] || endpoints.default;
  }

  async getProductAvailability(productId, storeId) {
    try {
      const response = await fetch(
        `${this.baseUrl}inventory/${storeId}/${productId}`,
        { headers: this.headers }
      );
      return await response.json();
    } catch (error) {
      console.error('Inventory check failed:', error);
      return { available: false, error: error.message };
    }
  }
}

// Enterprise-grade Order Processing System for Walmart/Costco
class AssistanceOrderProcessor {
  constructor(retailerConnection) {
    this.retailerAPI = retailerConnection;
    this.orderQueue = [];
    this.orderHistory = [];
    this.analyticsEnabled = true;
  }

  async submitOrder(orderData) {
    // Input validation
    this.validateOrderInput(orderData);
    
    // Add retailer-specific data
    const enhancedOrder = this.enhanceWithRetailerData(orderData);
    
    // Submit to queue and backend
    const orderRef = await this.persistOrder(enhancedOrder);
    
    // Track analytics if enabled for large retailers
    if (this.analyticsEnabled) {
      this.trackOrderAnalytics(enhancedOrder, 'submitted');
    }
    
    return orderRef;
  }

  enhanceWithRetailerData(orderData) {
    // Add retailer-specific data for Walmart/Costco
    const retailer = this.retailerAPI.getRetailerType();
    if (retailer === 'walmart') {
      orderData.departmentId = this.mapToDepartment(orderData.productCategory);
      orderData.walmartSpecific = {
        supercenterSection: this.getSectionMapping(orderData.productId),
        priorityLevel: this.calculatePriority(orderData)
      };
    } else if (retailer === 'costco') {
      orderData.warehouseZone = this.getWarehouseZone(orderData.productId);
      orderData.membershipTier = this.retailerAPI.getMembershipTier(orderData.userId);
    }
    return orderData;
  }
We designed the order form to be as simple as possible for customers, but the backend system is robust and enterprise-ready. The code shows our flexible architecture that can adapt to different retail environments like Walmart's supercenter layout or Costco's membership tiers and warehouse organization.

const StaffDashboard: React.FC = () => {
  const { staffId, staffRole, retailerType } = useStaffAuth();
  const [requests, setRequests] = useState([]);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    // Connect to real-time updates using WebSockets
    const socket = new WebSocket(
      `${RetailerConfig[retailerType].wsEndpoint}/staff/${staffId}`
    );
    
    socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.type === 'requests_update') {
        setRequests(data.requests);
        setLoading(false);
      }
    };
    
    return () => socket.close();
  }, [staffId, retailerType]);

from flask import Flask, request, jsonify
from flask_socketio import SocketIO, emit
import retail_integrations

socketio = SocketIO(app, cors_allowed_origins="*")

# Different retail integrations based on store type
retail_handlers = {
    'walmart': retail_integrations.WalmartIntegration(),
    'costco': retail_integrations.CostcoIntegration()
}

@app.route('/api/v1/assistance/assign', methods=['POST'])
def assign_assistance_request():
    data = request.json
    staff_id = data.get('staffId')
    request_id = data.get('requestId')
    store_id = data.get('storeId')
    
    # Get the correct retail integration
    retail_type = get_store_retail_type(store_id)
    retail_handler = retail_handlers[retail_type]
    
    # Update the request status in the backend system
    result = retail_handler.assign_staff_to_request(request_id, staff_id)
    
    # Notify all connected staff via WebSocket
    socketio.emit('request_assigned', result, room=f'store_{store_id}')
    
    return jsonify(result)
